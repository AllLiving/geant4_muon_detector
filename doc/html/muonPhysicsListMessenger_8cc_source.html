137 void muonPhysicsListMessenger::SetNewValue(G4UIcommand* command,
138  G4String newValue)
139 {
140  if( command == fSetAbsorptionCMD ) {
141  fPhysicsList->SetAbsorption(G4UIcmdWithABool::GetNewBoolValue(newValue));
142  }
143 
144  else if( command == fVerboseCmd ) {
145  fPhysicsList->SetVerbose(fVerboseCmd->GetNewIntValue(newValue));
146  }
147 
148  else if( command == fCerenkovCmd ) {
149  fPhysicsList->
150  SetNbOfPhotonsCerenkov(fCerenkovCmd->GetNewIntValue(newValue));
151  }
152 
153  else if (command == fPienuCMD) {
154  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
155  G4ParticleDefinition* particleDef = particleTable->FindParticle("pi+");
156  G4VDecayChannel* mode =
157  new G4PhaseSpaceDecayChannel("pi+",1.0,2,"e+","nu_e");
158  G4DecayTable* table = new G4DecayTable();
159  table->Insert(mode);
160  // mode = new G4PionRadiativeDecayChannel("pi+",0.000017);
161  // table->Insert(mode);
162  particleDef->SetDecayTable(table);
163  }
164 
165  else if (command == fPimunuCMD) {
166  G4ParticleTable* particleTable = G4ParticleTable::GetParticleTable();
167  G4ParticleDefinition* particleDef = particleTable->FindParticle("pi+");
168  G4VDecayChannel* mode =
169  new G4PhaseSpaceDecayChannel("pi+",1.000,2,"mu+","nu_mu");
170  G4DecayTable* table = new G4DecayTable();
171  table->Insert(mode);
172  particleDef->SetDecayTable(table);
173  }
174 
175  else if (command == fGammaCutCMD) {
176  fPhysicsList->SetCutForGamma(fGammaCutCMD
177  ->GetNewDoubleValue(newValue));
178  }
179  else if (command == fElectCutCMD) {
180  fPhysicsList->SetCutForElectron(fElectCutCMD
181  ->GetNewDoubleValue(newValue));
182  }
183  else if (command == fPosCutCMD) {
184  fPhysicsList->SetCutForPositron(fPosCutCMD
185  ->GetNewDoubleValue(newValue));
186  }
187  else if (command == fAllCutCMD) {
188  G4double cut = fAllCutCMD->GetNewDoubleValue(newValue);
189  fPhysicsList->SetCutForGamma(cut);
190  fPhysicsList->SetCutForElectron(cut);
191  fPhysicsList->SetCutForPositron(cut);
192  }
193  else if (command == fStepMaxCMD) {
194  fPhysicsList->SetStepMax(fStepMaxCMD
195  ->GetNewDoubleValue(newValue));
196  }
197  else if (command == fClearPhysicsCMD) {
198  fPhysicsList->ClearPhysics();
199  }
200  else if (command == fRemovePhysicsCMD) {
201  G4String name = newValue;
202  fPhysicsList->RemoveFromPhysicsList(name);
203  }
